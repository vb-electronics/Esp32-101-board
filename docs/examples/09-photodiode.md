=== "Примерен код"
    ```c++
    *
    В този пример измерената стойност от фототранзистора
    се показва на серийния монитор (Tools -> Serial Monitor).
    
    фототранзисторът (PT) e свързан към пин D34.
    */

    // константи, описващи използваните пинове
    const int PT_pin = 34;

    // променливи, използвани при четенето на фототранзистора
    int PT_value;

    void setup() {
    // настройване на серийната комуникация
    // Забележка: в серийният монитор трябва да бъда избрана същата скорост
    Serial.begin(115200);
    }

    void loop() {  
    // прочитане на стойността на фототранзистора (между 0 и 4095):
    PT_value = analogRead(PT_pin);
    
    // показване на стойностите на серийния монитор
    Serial.print("PT= ");
    Serial.print(PT_value);
    Serial.println("\t\t");

    delay(100);
    }

    ```
